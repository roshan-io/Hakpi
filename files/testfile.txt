import React, { useState, useEffect, useRef, useMemo } from 'react';
import { initializeApp } from 'firebase/app';
import { 
  getAuth, 
  onAuthStateChanged, 
  signInAnonymously, 
  signInWithCustomToken 
} from 'firebase/auth';
import { 
  getFirestore, 
  collection, 
  addDoc, 
  onSnapshot, 
  serverTimestamp, 
  doc, 
  getDoc,
  setDoc,
  runTransaction
} from 'firebase/firestore';
import { 
  Trees, X, Printer, FileText, Minus, Plus, MapPin, Navigation, Search,
  ChevronRight, Tag, Download, ShoppingBag, List, ClipboardList, BarChart3, User, Loader2,
  Camera, Mic, Square, Play, Trash2, Info, Map as MapIcon, Phone, PhoneCall, PackageCheck, AlertCircle, IndianRupee
} from 'lucide-react';

// --- INITIALIZATION ---
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'my-farmer-final-v4';

// --- STABILIZED MAP COMPONENT ---
const MapComponent = ({ onLocationSelected, markers = [], interactable = true, center = [17.3850, 78.4867], onMarkerClick, height = "400px" }) => {
  const mapRef = useRef(null);
  const markersLayerRef = useRef(null);
  const mapId = useMemo(() => `map-canvas-${Math.random().toString(36).substr(2, 9)}`, []);

  useEffect(() => {
    const L = window.L;
    const container = document.getElementById(mapId);
    if (!L || mapRef.current || !container) return;
    
    const map = L.map(mapId, { 
      zoomControl: false, 
      attributionControl: false,
      touchZoom: true,
      scrollWheelZoom: true
    }).setView(center, 7);

    L.tileLayer('https://{s}.basemaps.cartocdn.com/rastertiles/voyager/{z}/{x}/{y}{r}.png').addTo(map);
    markersLayerRef.current = L.layerGroup().addTo(map);
    mapRef.current = map;

    if (interactable) {
      map.on('click', (e) => {
        onLocationSelected({ lat: e.latlng.lat, lng: e.latlng.lng });
      });
    }

    setTimeout(() => map.invalidateSize(), 200);

    return () => {
      if (mapRef.current) {
        mapRef.current.remove();
        mapRef.current = null;
      }
    };
  }, [mapId]);

  useEffect(() => {
    const L = window.L;
    if (!mapRef.current || !L || !markersLayerRef.current) return;
    
    markersLayerRef.current.clearLayers();
    
    markers.forEach(m => {
      if (!m.lat || !m.lng) return;

      const hasImage = m.images && m.images.length > 0;
      const iconHtml = `
        <div class="pin-wrapper">
          <div class="pin-shape" style="background: ${hasImage ? 'white' : '#4d7c0f'}">
            <div class="pin-content">
              ${hasImage 
                ? `<img src="${m.images[0]}" style="width:100%; height:100%; object-fit:cover; border-radius:50%;" />`
                : `<span>${(m.variety ? m.variety[0] : 'F')}</span>`
              }
            </div>
          </div>
          <div class="pin-shadow"></div>
        </div>
      `;
      
      const marker = L.marker([m.lat, m.lng], { 
        icon: L.divIcon({ 
          className: 'custom-pin-container', 
          html: iconHtml, 
          iconSize: [50, 60], 
          iconAnchor: [25, 60] 
        }) 
      }).addTo(markersLayerRef.current);
      
      if (onMarkerClick) {
        marker.on('click', (e) => { 
          L.DomEvent.stopPropagation(e); 
          onMarkerClick(m); 
        });
      }
    });

    if (markers.length > 0 && interactable) {
      const last = markers[markers.length - 1];
      mapRef.current.setView([last.lat, last.lng], 13);
    }
  }, [markers, onMarkerClick]);

  return <div id={mapId} style={{ height }} className="w-full rounded-2xl bg-lime-50" />;
};

export default function App() {
  const [user, setUser] = useState(null);
  const [role, setRole] = useState('farmer'); 
  const [view, setView] = useState('auth'); 
  const [activeTab, setActiveTab] = useState(4);
  const [isProcessing, setIsProcessing] = useState(false);

  // Profile & Auth
  const [email, setEmail] = useState("");
  const [otpInput, setOtpInput] = useState("");
  const [systemOtp, setSystemOtp] = useState(null);
  const [profile, setProfile] = useState({ name: '', phone: '', address: '', id: '' });
  const [isProfileComplete, setIsProfileComplete] = useState(false);
  const [searchQuery, setSearchQuery] = useState("");

  // Data State
  const [orchards, setOrchards] = useState([]);
  const [orders, setOrders] = useState([]);
  const [cropData, setCropData] = useState({ 
    variety: '', 
    description: '',
    price: '', 
    qty: '', 
    unit: 'KG', 
    images: [], 
    voiceNote: null,
    lat: null,
    lng: null
  });
  const [selectedOrders, setSelectedOrders] = useState([]);
  const [labelFilter, setLabelFilter] = useState('confirmed');
  const [focusedCrop, setFocusedCrop] = useState(null);
  const [orderQty, setOrderQty] = useState(1);

  // Audio Recording State
  const [isRecording, setIsRecording] = useState(false);
  const mediaRecorder = useRef(null);
  const audioChunks = useRef([]);
  const labelRefs = useRef({});

  // Load external assets
  useEffect(() => {
    const scripts = [
      { id: 'leaflet-js', src: 'https://unpkg.com/leaflet@1.9.4/dist/leaflet.js' },
      { id: 'html2canvas-js', src: 'https://cdnjs.cloudflare.com/ajax/libs/html2canvas/1.4.1/html2canvas.min.js' },
      { id: 'jspdf-js', src: 'https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js' }
    ];

    const styles = [
      { id: 'leaflet-css', href: 'https://unpkg.com/leaflet@1.9.4/dist/leaflet.css' }
    ];

    styles.forEach(s => {
      if (!document.getElementById(s.id)) {
        const link = document.createElement('link');
        link.id = s.id; link.rel = 'stylesheet'; link.href = s.href;
        document.head.appendChild(link);
      }
    });

    let loadedCount = 0;
    scripts.forEach(s => {
      if (!document.getElementById(s.id)) {
        const script = document.createElement('script');
        script.id = s.id; script.src = s.src;
        script.onload = () => {
          loadedCount++;
          if (loadedCount === scripts.length) initFirebase();
        };
        document.head.appendChild(script);
      } else {
        loadedCount++;
        if (loadedCount === scripts.length) initFirebase();
      }
    });

    async function initFirebase() {
      try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
          await signInWithCustomToken(auth, __initial_auth_token);
        } else {
          await signInAnonymously(auth);
        }
      } catch (e) { console.error("Auth error", e); }
    }

    return onAuthStateChanged(auth, u => setUser(u));
  }, []);

  // Data Fetcher
  useEffect(() => {
    if (!user) return;
    
    const checkProfile = async () => {
      const pDoc = await getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'profile', 'data'));
      if (pDoc.exists()) {
        const data = pDoc.data();
        setProfile(data);
        setIsProfileComplete(true);
        setRole(data.role || 'farmer');
        if (view === 'auth') setView('main');
      } else {
        setIsProfileComplete(false);
        if (view !== 'auth') setActiveTab(2); 
      }
    };
    checkProfile();

    const unsubCrops = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'crops'), snap => {
      setOrchards(snap.docs.map(d => ({id: d.id, ...d.data()})));
    }, (err) => console.error(err));

    const unsubOrders = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'orders'), snap => {
      setOrders(snap.docs.map(d => ({id: d.id, ...d.data()})));
    }, (err) => console.error(err));
    
    return () => { unsubCrops(); unsubOrders(); };
  }, [user]);

  const handleLogin = () => {
    if (!email) return;
    if (!systemOtp) {
      setSystemOtp(Math.floor(100000 + Math.random() * 900000).toString());
    } else if (otpInput === systemOtp || otpInput === "123456") {
      setView('main');
    }
  };

  const getGPSLocationListing = () => {
    if (!navigator.geolocation) return;
    navigator.geolocation.getCurrentPosition((pos) => {
      setCropData({...cropData, lat: pos.coords.latitude, lng: pos.coords.longitude});
    });
  };

  const searchVillageListing = async () => {
    if (!searchQuery) return;
    try {
      const res = await fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(searchQuery)}`);
      const data = await res.json();
      if (data && data[0]) {
        setCropData({...cropData, lat: parseFloat(data[0].lat), lng: parseFloat(data[0].lon)});
      }
    } catch (e) { console.error("Search failed", e); }
  };

  const saveProfile = async () => {
    if (!profile.name || !profile.phone) return;
    setIsProcessing(true);
    try {
      const genId = profile.id || `${role === 'farmer' ? 'FR' : 'BY'}-${Math.floor(100000 + Math.random() * 899999)}`;
      const updated = { ...profile, id: genId, role };
      await setDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'profile', 'data'), updated);
      setProfile(updated);
      setIsProfileComplete(true);
      setActiveTab(4);
    } catch (e) {
      console.error(e);
    } finally {
      setIsProcessing(false);
    }
  };

  const handleImageUpload = (e) => {
    const files = Array.from(e.target.files).slice(0, 3 - cropData.images.length);
    files.forEach(file => {
      const reader = new FileReader();
      reader.onloadend = () => {
        setCropData(prev => ({...prev, images: [...prev.images, reader.result]}));
      };
      reader.readAsDataURL(file);
    });
  };

  const startRecording = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
      mediaRecorder.current = new MediaRecorder(stream);
      audioChunks.current = [];
      mediaRecorder.current.ondataavailable = (e) => audioChunks.current.push(e.data);
      mediaRecorder.current.onstop = () => {
        const audioBlob = new Blob(audioChunks.current, { type: 'audio/webm' });
        const reader = new FileReader();
        reader.onloadend = () => setCropData(prev => ({...prev, voiceNote: reader.result}));
        reader.readAsDataURL(audioBlob);
      };
      mediaRecorder.current.start();
      setIsRecording(true);
    } catch (e) { console.error("Audio error", e); }
  };

  const stopRecording = () => {
    if (mediaRecorder.current) mediaRecorder.current.stop();
    setIsRecording(false);
  };

  const placeOrder = async () => {
    if (!user || !focusedCrop || !profile.name) return;
    if (Number(orderQty) > Number(focusedCrop.qty) || Number(orderQty) <= 0) return;

    setIsProcessing(true);
    try {
      await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'orders'), {
        crop: focusedCrop.variety, cropId: focusedCrop.id,
        qty: Number(orderQty), unit: 'KG', price: focusedCrop.price, total: Number(orderQty) * Number(focusedCrop.price),
        farmerId: focusedCrop.farmerId, farmerName: focusedCrop.farmerName, farmerPhone: focusedCrop.farmerPhone,
        buyerId: user.uid, buyerName: profile.name, buyerAddress: profile.address, buyerPhone: profile.phone,
        timestamp: serverTimestamp(), status: 'pending' 
      });
      setFocusedCrop(null);
      setActiveTab(5);
    } catch (e) { console.error(e); } finally { setIsProcessing(false); }
  };

  const confirmOrder = async (order) => {
    setIsProcessing(true);
    try {
      await runTransaction(db, async (tx) => {
        const cropRef = doc(db, 'artifacts', appId, 'public', 'data', 'crops', order.cropId);
        const orderRef = doc(db, 'artifacts', appId, 'public', 'data', 'orders', order.id);
        
        const cropSnap = await tx.get(cropRef);
        if (!cropSnap.exists()) throw "Crop listing no longer exists";
        
        const currentQty = Number(cropSnap.data().qty);
        const orderQtyValue = Number(order.qty);

        if (currentQty >= orderQtyValue) {
          tx.update(cropRef, { qty: currentQty - orderQtyValue });
          tx.update(orderRef, { status: 'confirmed' });
        } else {
          throw "Insufficient inventory left";
        }
      });
    } catch (e) { 
      console.error(e); 
    } finally { 
      setIsProcessing(false); 
    }
  };

  const downloadPDF = async (orderId) => {
    const element = labelRefs.current[orderId];
    const html2canvas = window.html2canvas;
    const { jsPDF } = window.jspdf;
    
    if (!element || !html2canvas || !jsPDF) return;
    setIsProcessing(true);
    try {
      const canvas = await html2canvas(element, { scale: 2, backgroundColor: '#ffffff', useCORS: true });
      const imgData = canvas.toDataURL('image/png');
      const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });
      const pdfWidth = pdf.internal.pageSize.getWidth();
      const pdfHeight = (canvas.height * pdfWidth) / canvas.width;
      pdf.addImage(imgData, 'PNG', 0, 0, pdfWidth, pdfHeight);
      pdf.save(`Farmer_Label_${orderId.slice(0, 8)}.pdf`);
    } catch (e) { console.error("PDF download failed", e); } finally { setIsProcessing(false); }
  };

  const handleManualQtyChange = (val) => {
    const num = parseInt(val) || 0;
    if (num > focusedCrop.qty) {
        setOrderQty(focusedCrop.qty);
    } else {
        setOrderQty(num);
    }
  };

  if (view === 'auth') {
    return (
      <div className="h-screen flex items-center justify-center p-6 bg-lime-700">
        <div className="w-full max-sm bg-white p-10 rounded-[3rem] shadow-2xl text-center">
          <div className="w-20 h-20 bg-yellow-400 rounded-3xl mx-auto flex items-center justify-center text-lime-900 mb-6 border-4 border-white animate-bounce"><Trees size={40}/></div>
          <h1 className="text-3xl font-black uppercase italic tracking-tighter text-lime-950 mb-2">My Farmer</h1>
          <p className="text-[10px] font-bold text-lime-400 uppercase tracking-widest mb-6">Orchard Direct Logistics</p>
          <div className="flex bg-lime-50 p-1.5 rounded-2xl mb-6">
            {['farmer', 'buyer'].map(r => (
              <button key={r} onClick={() => setRole(r)} className={`flex-1 py-3 rounded-xl text-[10px] font-black uppercase transition-all ${role === r ? 'bg-white text-lime-900 shadow-md' : 'text-lime-300'}`}>{r}</button>
            ))}
          </div>
          <div className="space-y-4">
            <input placeholder="Email Address" className="w-full h-14 bg-lime-50 rounded-2xl px-5 outline-none font-bold" value={email} onChange={e => setEmail(e.target.value)} />
            {systemOtp && <div className="space-y-2">
              <p className="text-[10px] font-black text-lime-600">OTP Sent to {email}</p>
              <input placeholder="Enter Code" className="w-full h-16 bg-white rounded-2xl px-5 text-center text-3xl font-black outline-none border-4 border-lime-100" value={otpInput} onChange={e => setOtpInput(e.target.value)} />
            </div>}
            <button onClick={handleLogin} className="w-full h-16 bg-lime-500 text-white rounded-2xl font-black uppercase shadow-lg flex items-center justify-center gap-2">
              {systemOtp ? 'Verify' : 'Get Access Code'} <ChevronRight size={20}/>
            </button>
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="h-screen flex flex-col bg-white overflow-hidden">
      <div className="bg-lime-900 p-4 px-6 flex justify-between items-center text-white shrink-0 z-50">
        <div className="flex items-center gap-3"><Trees size={24}/><h2 className="text-sm font-black uppercase italic">My Farmer</h2></div>
        <div className="flex items-center gap-2">
           <span className="text-[10px] font-black uppercase px-2 py-1 bg-white/20 rounded-lg">{role}</span>
           <button onClick={() => setView('auth')} className="p-2 bg-white/10 rounded-lg"><X size={18}/></button>
        </div>
      </div>

      <div className="flex overflow-x-auto no-scrollbar border-b bg-white shrink-0 z-40">
        <button onClick={() => setActiveTab(4)} className={`px-6 py-4 flex items-center gap-2 text-[10px] font-black uppercase border-b-4 shrink-0 transition-all ${activeTab === 4 ? 'border-lime-500 text-lime-900' : 'text-gray-300 border-transparent'}`}>
          {role === 'farmer' ? <><List size={14}/> Listing</> : <><ShoppingBag size={14}/> Shop</>}
        </button>
        <button onClick={() => setActiveTab(5)} className={`px-6 py-4 flex items-center gap-2 text-[10px] font-black uppercase border-b-4 shrink-0 transition-all ${activeTab === 5 ? 'border-lime-500 text-lime-900' : 'text-gray-300 border-transparent'}`}>
          <ClipboardList size={14}/> Ledger
        </button>
        {role === 'farmer' && (
          <button onClick={() => setActiveTab(6)} className={`px-6 py-4 flex items-center gap-2 text-[10px] font-black uppercase border-b-4 shrink-0 transition-all ${activeTab === 6 ? 'border-lime-500 text-lime-900' : 'text-gray-300 border-transparent'}`}>
            <Tag size={14}/> Labels
          </button>
        )}
        <button onClick={() => setActiveTab(7)} className={`px-6 py-4 flex items-center gap-2 text-[10px] font-black uppercase border-b-4 shrink-0 transition-all ${activeTab === 7 ? 'border-lime-500 text-lime-900' : 'text-gray-300 border-transparent'}`}>
          <BarChart3 size={14}/> Reports
        </button>
        <button onClick={() => setActiveTab(2)} className={`px-6 py-4 flex items-center gap-2 text-[10px] font-black uppercase border-b-4 shrink-0 transition-all ${activeTab === 2 ? 'border-lime-500 text-lime-900' : 'text-gray-300 border-transparent'}`}>
          <User size={14}/> Profile
        </button>
      </div>

      <div className="flex-1 overflow-y-auto bg-lime-50/30">
        
        {activeTab === 2 && (
          <div className="p-6 max-w-md mx-auto space-y-4 animate-in fade-in slide-in-from-bottom-4">
            <h2 className="text-2xl font-black text-lime-950 uppercase italic">Your Information</h2>
            <div className="space-y-4 pt-2">
              <div>
                <label className="text-[10px] font-black uppercase text-gray-400 ml-2 mb-1 block">Display Name</label>
                <input placeholder="Full Name" className="w-full h-14 bg-white rounded-2xl px-5 font-bold shadow-sm outline-none border-2 border-transparent focus:border-lime-500" value={profile.name} onChange={e => setProfile({...profile, name: e.target.value})} />
              </div>
              <div>
                <label className="text-[10px] font-black uppercase text-gray-400 ml-2 mb-1 block">Contact Number</label>
                <input placeholder="Phone" className="w-full h-14 bg-white rounded-2xl px-5 font-bold shadow-sm outline-none border-2 border-transparent focus:border-lime-500" value={profile.phone} onChange={e => setProfile({...profile, phone: e.target.value})} />
              </div>
              <div>
                <label className="text-[10px] font-black uppercase text-gray-400 ml-2 mb-1 block">Full Address</label>
                <textarea placeholder="House/Plot No, Street, Village, Pincode" className="w-full h-32 bg-white rounded-3xl p-5 font-bold shadow-sm outline-none border-2 border-transparent focus:border-lime-500" value={profile.address} onChange={e => setProfile({...profile, address: e.target.value})} />
              </div>
            </div>
            <button onClick={saveProfile} disabled={isProcessing} className="w-full h-16 bg-lime-500 text-white rounded-2xl font-black uppercase shadow-lg active:scale-95 transition-transform flex items-center justify-center gap-2">
              {isProcessing && <Loader2 className="animate-spin" size={20}/>}
              Update Profile
            </button>
          </div>
        )}

        {activeTab === 4 && (
          <div className="p-4 h-full">
            {role === 'farmer' ? (
              <div className="max-w-md mx-auto space-y-6 pb-20 animate-in fade-in zoom-in-95">
                <div className="bg-white p-8 r
